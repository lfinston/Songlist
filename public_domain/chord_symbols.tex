\vskip2\baselineskip
\A{\hbox{\vbox to 0pt{\vskip1.5pt\hbox to 0pt{\hskip1.8pt ${}_\circ$\hss}\vss}}C}{C alone, i.e., not a chord (C)}
%% Major
\A{C}{C major (C-E-G)}
\A{$\rm{C}_{4}$}{C suspended $4^{\rm{th}}$ (C-F-G)}
\A{$\rm{C}_4$\lower1.5pt\hbox{-}\hskip5pt ${}_3$}{C suspended $4^{\rm{th}}$ resolving to $3^{\rm{rd}}$ (C-F-G -- C-E-G)}
\A{$\rm{C}_4$\lower1.5pt\hbox{-}\hskip5pt $\rm{C}_3$}{ibid.}
\A{C/E}{C major with E in bass (E-C-G or E-G-C)}
\A{$\rm{C}^6$}{C sixth (C-E-G-A)}
\A{$\rm{C}_9$}{C add ninth (C-E-G-D)}
\A{$\rm{C}^6_9$}{C six-nine (C-E-G-A-D)}
\A{$\rm{C}_{\Slash{5}}$}{C major without $5^{\rm{th}}$ (C-E)}
\A{$\rm{C}^\triangle$}{C major $7^{\rm{th}}$ (C-E-G-B)}
\A{$\rm{C}^\triangle_9$}{C major $7^{\rm{th}}$ with added $9^{\rm{th}}$ (C-E-G-B-D)}
\A{$\rm{C}$\raise4pt\hbox{{\SMALL +}}}{C augmented (C-E-G$\sharp$)}
\A{$\rm{C}^7$}{C dominant $7^{\rm{th}}$ (C-E-G-B$\flat$)}
\A{$\rm{C}^9$}{C dominant $9^{\rm{th}}$ (C-E-G-B$\flat$-D)}
\A{$\rm{C}^{11}$}{C dominant $11^{\rm{th}}$ (C-F-G-B$\flat$-D)}
\A{$\rm{C}^{13}$}{C dominant $13^{\rm{th}}$ (C-E-G-B$\flat$-D-A)}
\A{$\rm{C}^7_{\flat 9}$}{C dominant $7^{\rm{th}}$ with flat $9^{\rm{th}}$(C-E-G-B$\flat$-D$\flat$)}
\A{$\rm{C}^7$\raise4pt\hbox{{\SMALL +}}}{C dominant $7^{\rm{th}}$ with augmented $5^{\rm{th}}$ (C-E-G$\sharp$-B$\flat$)}
\A{$\rm{C}^9_{\Slash{3}\Slash{5}}$}{C dominant $9^{\rm{th}}$ without $3^{\rm{rd}}$ and $5^{\rm{th}}$ (C-B$\flat$-D)}
\A{$\rm{C}^9$\raise4pt\hbox{{\SMALL +}}}{C dominant $9^{\rm{th}}$ with augmented $5^{\rm{th}}$ (C-E-G$\sharp$-B$\flat$-D)}
\A{$\rm{Cm}^\triangle$}{C minor with major $7^{\rm{th}}$ (C-E$\flat$-G-B)}
\A{$\rm{C}^\triangle$}{C major $7^{\rm{th}}$ (C-E-G-B)}
\A{$\rm{Cm}^\triangle$}{C minor with major $7^{\rm{th}}$ (C-E$\flat$-G-B)}
\A{$\rm{C}^\circ$}{C diminished $7^{\rm{th}}$ (C-E$\flat$-G$\flat$-B$\flat\flat$)}
\A{$\rm{C}^\circ_{\Slash{7}}$}{C diminished triad, i.e., without diminished $7^{\rm{th}}$ (C-E$\flat$-G$\flat$)}
\A{$\rm{C}^{\halfdimin}$}{C half-diminished $7^{\rm{th}}$ (C-E$\flat$-G$\flat$-B$\flat$)}
