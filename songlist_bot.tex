\xdef\song#1#2#3#4#5#6#7#8#9{%
\hbox to 7cm{\vbox{\valign{%
%% 1. Song title
\vrule height 12pt depth 2pt width 0pt {\largebx ##}\vskip2pt&
%% 2. Lead Sheet
\vrule height 12pt depth 2pt width 0pt \hskip1em
\hbox to 5cm{##\leaders\copy\dotbox\hss}#2&
%% 3. Score
\vrule height 12pt depth 2pt width 0pt \hskip1em\hbox to 5cm{##\leaders\copy\dotbox\hss}#3&
%% 4. Score:  Guit. Mel.
\vrule height 12pt depth 2pt width 0pt \hskip1em\hbox to 5cm{##\leaders\copy\dotbox\hss}#4&
%% 5. Arr. Solo Guit.
\vrule height 12pt depth 2pt width 0pt \hskip1em\hbox to 5cm{##\leaders\copy\dotbox\hss}#5&
%% 6. Recordings
\vrule height 12pt depth 2pt width 0pt \hskip1em\hbox to 5cm{##\leaders\copy\dotbox\hss}#6&
%% 7. Author(s)
\vrule height 12pt depth 12pt width 0pt \hskip1em 
## \hbox to 2cm{\vtop to 0pt{#7\vss}\hss}& 
%% 8. Copyright
\vrule height 12pt depth 2pt width 0pt \hskip1em 
## #8\vtop to 18pt{\vfil}&
%%
%% 9. Notes
\vrule height 12pt depth 2pt width 0pt \hskip1em ## 
\hbox to 2cm{\vtop to 0pt{#9\vss}\hss} \cr
%% End of template
#1&Lead Sheet&Score&%
Score: Guit.~Mel.&Arr.~Solo Guit.:&Recordings:&Author(s):&Copyright:%
&{\largebx Notes:}\cr}\vskip\notesskip\vss}\hss}}
